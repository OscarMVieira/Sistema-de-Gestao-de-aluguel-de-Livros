CREATE TABLE requisicoes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT, -- Liga ao id da tabela users
    livro_id INT, -- Liga ao id da tabela livros
    data_pedido DATE NOT NULL,
    data_levantamento DATE DEFAULT NULL,
    data_devolucao DATE DEFAULT NULL,
    estado VARCHAR(50) DEFAULT 'Pendente', 
    observacao TEXT,
    -- Criar as ligações (Constraints)
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (livro_id) REFERENCES livros(ID_Livro) ON DELETE CASCADE
);
